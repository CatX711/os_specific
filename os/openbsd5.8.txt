~OpenBSD5.8~

## a few of the basic settings for openbsd 5.8
## after os installation, login as root (or 'su') and add user to wheel:

usermod -G wheel ~username~

# OR

wheel:*:0:root,~user~

## enable ~doas~

vi /etc/doas.conf

## and add:

permit keepenv { PKG_PATH ENV PS1 SSH_AUTH_SOCK } :wheel

## SAVE

## add a repository from the internet

export PKG_PATH=http://www.mirrorservice.org/pub/OpenBSD/5.8/packages/amd64/

## or use the CD

export PKG_PATH=/mnt/cdrom/$(uname -r)/packages/$(machine -a)/

## as root add to ~/.profile

export PKG_PATH=http://www.mirrorservice.org/pub/OpenBSD/$(uname -r)/packages/$(machine -a)

## to install programs:

## become 'root' using 'su':

pkg_add ~program~

## or using 'doas':

doas pkg_add ~program~
